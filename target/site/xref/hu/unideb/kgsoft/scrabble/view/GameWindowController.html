301     public void computerThread(final Computer computer, final Gameboard gboard,
302             final Dictionary dict) {
303         Task<Void> task = new Task<Void>() {
304             @Override
305             protected Void call() throws Exception {
306                 System.out.println("Computer thread.");
307                 computer.play(gboard, dict);
308                 return null;
309             }
310         };
311         task.stateProperty().addListener(new ChangeListener<Worker.State>() {
312 
313             @Override
314             public void changed(ObservableValue<? extends State> observable,
315                     State oldValue, State newValue) {
316                 if (newValue == Worker.State.SUCCEEDED) {
317                     System.out.println("Computer thread succeeded.");
318                     mainCtr.computerReady();
319                 }
320             }
321         });
322 
323         new Thread(task).start();
324     }

327         Task<Void> task = new Task<Void>() {

336 
337             @Override

341                     System.out.println("Game saved.");
342                     mainCtr.gameSaved();
343                 }
344             }
345         });
346 
347         new Thread(task).start();
348     }

399     @Override
400     public void updateGameStateInfo(GameState gameState) {
401         // doneButton enable/disable
402         if (doneButton.disableProperty().get()) {
403             if (gameState.isBoardLegal() && !gameState.isTileInHand()
404                     && gameState.isWordsCorrect() && gameState.isPlayersTurn()) {
405                 doneButton.setDisable(false);
406             }
407         } else if (!gameState.isBoardLegal() || gameState.isTileInHand()
408                 || !gameState.isWordsCorrect() || !gameState.isPlayersTurn()) {
409             doneButton.setDisable(true);
410         }
411 
412         // redrawButton enable/disable
413         if (redrawButton.disabledProperty().get()) {
414             if (!gameState.isMovableTiles() && gameState.getTilesInBag() > 0
415                     && gameState.isPlayersTurn() && gameState.isGameStarted()) {
416                 redrawButton.setDisable(false);
417             }
418         } else if (gameState.isMovableTiles() || gameState.getTilesInBag() == 0
419                 || !gameState.isPlayersTurn() || !gameState.isGameStarted()) {
420             redrawButton.setDisable(true);
421         }
422 
423         // passButton enable/disable
424         if (passButton.disabledProperty().get()) {
425             if (!gameState.isMovableTiles() && gameState.isPlayersTurn() && gameState.isGameStarted()) {
426                 passButton.setDisable(false);
427             }
428         } else if (gameState.isMovableTiles() || !gameState.isPlayersTurn() || !gameState.isGameStarted()) {
429             passButton.setDisable(true);
430         }
431 
432         // Show Player and Computer names and scores
433         if (gameState.isPlayersTurn() && gameState.isGameStarted()) {
434             playerNameLabel.setText(gameState.getPlayerName() + " <--");
435             computerNameLabel.setText("Szmtgp");
436         } else if (gameState.isGameStarted()) {
437             playerNameLabel.setText(gameState.getPlayerName());
438             computerNameLabel.setText("Szmtgp <--");
439         } else {
440             playerNameLabel.setText(gameState.getPlayerName());
441             computerNameLabel.setText("Szmtgp");
442         }
443         playerScoreLabel.setText(Integer.toString(gameState.getPlayerScore()));
444         computerScoreLabel.setText(Integer.toString(gameState
445                 .getComputerScore()));
446 
447         // Show last words.
448         if (gameState.getLastWords() != null) {
449             if (gameState.getLastWords().size() > 1) {
450                 lastWordOrWordsLabel.setText("Legutbbi szavak:");
451             } else {
452                 lastWordOrWordsLabel.setText("Legutbbi sz:");
453             }
454 
455             StringJoiner sjLastWords = new StringJoiner(", ");
456             for (String w : gameState.getLastWords()) {
457                 sjLastWords.add(w);
458             }
459             lastWordOrWordsListingLabel.setText(sjLastWords.toString());
460             lastWordOrWordsPointsLabel.setText(Integer.toString(gameState
461                     .getLastWordsPoints()));
462         }
463 
464         // Show current words.
465         if (gameState.getCurrentWords() != null) {
466             if (gameState.getCurrentWords().size() > 1) {
467                 currWordOrWordsLabel.setText("ppen kirakott szavak:");
468             } else {
469                 currWordOrWordsLabel.setText("ppen kirakott sz:");
470             }
471         }
472 
473         if (gameState.isBoardLegal()) {
474             StringJoiner sjCurrWords = new StringJoiner(", ");
475             for (String w : gameState.getCurrentWords()) {
476                 sjCurrWords.add(w);
477             }
478             if (gameState.isWordsCorrect()) {
479                 currWordOrWordsListingLabel.setTextFill(Color.BLACK);
480                 currWordOrWordsListingLabel.setText(sjCurrWords.toString());
481                 currWordOrWordsPointsLabel.setText(Integer.toString(gameState
482                         .getCurrentWordsPoints()));
483             } else {
484                 currWordOrWordsListingLabel.setTextFill(Color.RED);
485                 currWordOrWordsListingLabel.setText(sjCurrWords.toString());
486                 currWordOrWordsPointsLabel.setText("");
487             }
488         } else {
489             currWordOrWordsListingLabel.setText("");
490             currWordOrWordsPointsLabel.setText("");
491         }
492 
493         // Show computer is thinking message.
494         if (gameState.isPlayersTurn() || !gameState.isGameStarted()) {
495             if (computerThinkingLabel.isVisible()) {
496                 computerThinkingLabel.setVisible(false);
497             }
498             if (computerThinkingIndicator.isVisible()) {
499                 computerThinkingIndicator.setVisible(false);
500             }
501         } else {
502             if (!computerThinkingLabel.isVisible()) {
503                 computerThinkingLabel.setVisible(true);
504             }
505             if (!computerThinkingIndicator.isVisible()) {
506                 computerThinkingIndicator.setVisible(true);
507             }
508         }
509

568         alert.setTitle("Megersts");
569         alert.setHeaderText(header);
570         alert.setContentText(text);
571 
572         alert.setHeight(300);
573         
574         ButtonType buttonTypeOK = new ButtonType(okText);        
575         ButtonType buttonTypeCancel = new ButtonType("Mgsem", ButtonData.CANCEL_CLOSE);
576 
577         alert.getButtonTypes().setAll(buttonTypeOK, buttonTypeCancel);
578 
579         Optional<ButtonType> result = alert.showAndWait();
580         if (result.get() == buttonTypeOK) {
581             return true;        
582         } else {
583             return false;
584         }
