16      private Stage primaryStage;

22      /**
23       * Sets the reference to the <code>Controller</code> object.

26      public static void setMainCtr(Controller mainCtr) {

30      @Override
31      public void start(Stage primaryStage) throws Exception {
32          this.primaryStage = primaryStage;
33          this.primaryStage.setTitle("Bejelentkez√©s");
34          
35          showLoginWindow();      
36      }
37      
38      public void initRootLayout() {
39          try {
40              FXMLLoader loader = new FXMLLoader();
41              loader.setLocation(ScrabbleApp.class.getResource("/fxmls/RootLayout.fxml"));
42              loader.setController(new MenuBarController());
43              rootLayout = (BorderPane) loader.load();
44              
45              Scene scene = new Scene(rootLayout);
46              primaryStage.setScene(scene);
47              primaryStage.show();            
48              primaryStage.setResizable(false);
49              
50              MenuBarController menuBarController = loader.getController();
51              menuBarController.setMainCtr(mainCtr);
52              menuBarController.setMainApp(this);
53              mainCtr.setMenuBarController(menuBarController);
54          } catch (IOException e) {
55              e.printStackTrace();
56          }
57      }
58      
59      public void showScrabbleGameWindow() {
60          try {
61              FXMLLoader loader = new FXMLLoader();
62              loader.setLocation(ScrabbleApp.class.getResource("/fxmls/ScrabbleGameWindow.fxml"));
63              loader.setController(new GameWindowController());
64              AnchorPane scrabbleGameWindow = (AnchorPane) loader.load();
65              
66              rootLayout.setCenter(scrabbleGameWindow);
67              
68              GameWindowController windowCtr = loader.getController();
69              windowCtr.setMainCtr(mainCtr);
70              mainCtr.setWindowCtr(windowCtr);
71              
72          } catch (IOException e) {
73              e.printStackTrace();
74          }
75      }
76      
77      public void showLoginWindow() {
78          try {
79              FXMLLoader loader = new FXMLLoader();
80              loader.setLocation(this.getClass().getResource("/fxmls/LoginWindow.fxml"));
81              loader.setController(new LoginWindowController());
82              AnchorPane loginWindow = (AnchorPane) loader.load();
83              
84              Scene scene = new Scene(loginWindow);
85              primaryStage.setScene(scene);
86              primaryStage.show();
87              
88              primaryStage.setResizable(false);
89              
90              LoginWindowController loginCtr = loader.getController();
91              loginCtr.setMainApp(this);
92              loginCtr.setMainCtr(mainCtr);
93              mainCtr.setLoginController(loginCtr);            
94          } catch (IOException e) {
95              e.printStackTrace();
96          }
97      }
98      
99      public void switchToGameWindow() {
100         primaryStage.setTitle("Scrabble");
101         
102         initRootLayout();
103         showScrabbleGameWindow();
104     }
105     
106     public Stage getPrimaryStage() {
107         return primaryStage;
