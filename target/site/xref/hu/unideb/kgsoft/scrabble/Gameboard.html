108     public Gameboard(Field[][] fields) {
109         board = fields;
110         movableTiles = 0;
111         for (int i = 0; i < BOARD_SIZE; i++) {
112             for (int j = 0; j < BOARD_SIZE; j++) {
113                 if (board[i][j].getStatus() == Field.Status.MOVABLE) {
114                     movableTiles++;
115                 }
116             }
117         }
118     }

120     /**
121      * If it is possible, places a tile at the given position on the board, with
122      * a given tile code. If the tile is a joker, use the
123      * <code>setJokerTile</code> method instead.
124      * 
125      * @param row
126      *            the row index where the tile should be placed
127      * @param col
128      *            the column index where the tile should be placed
129      * @param code
130      *            the code of the tile
131      * @return <code>true</code> if the tile has been successfully placed on the
132      *         board, <code>false</code> otherwise
133      */
134     public boolean setTile(int row, int col, int code) {
135         if (board[row][col].getStatus() == Field.Status.EMPTY) {
136             board[row][col].setTileCode(code);
137             board[row][col].setStatus(Field.Status.MOVABLE);
138             movableTiles++;
139             return true;
140         } else {
141             return false;
142         }
143     }

146      * If it is possible, places a joker at the given position, replacing the
147      * given tile code.
148      * 
149      * @param row
150      *            the row index where the tile should be placed
151      * @param col
152      *            the column index where the tile should be placed
153      * @param code
154      *            the code of the tile that is replaced by the joker
155      * @return <code>true</code> if the tile has been successfully placed on the

158     public boolean setJokerTile(int row, int col, int code) {
159         if (board[row][col].getStatus() == Field.Status.EMPTY) {
160             board[row][col].setTileCode(Letters.JOKER_CODE);
161             board[row][col].setJokerTileCode(code);
162             board[row][col].setStatus(Field.Status.MOVABLE);
163             movableTiles++;
164             return true;
165         } else {
166             return false;
167         }

170     /**
171      * Returns the tile code of the field with the given row and column indices,
172      * if the tile is movable, regarding the rules, and sets its value to
173      * {@code Field.NOTILE}.
174      * 
175      * @param row
176      *            the row index from where the tile should be removed
177      * @param col
178      *            the column index from where the tile should be removed
179      * @return the tile code of the tile picked up, or the
180      *         <code>Field.NOTILE</code> constant if there was no tile on the
181      *         field
182      */
183     public int pickUpTile(int row, int col) {
184         if (board[row][col].getStatus() == Field.Status.MOVABLE) {
185             int code = board[row][col].getTileCode();
186             board[row][col].setTileCode(Field.NOTILE);
187             board[row][col].setJokerTileCode(Field.NOTILE);
188             board[row][col].setStatus(Field.Status.EMPTY);
189             movableTiles--;
190             return code;
191         } else {
192             return Field.NOTILE;
193         }
194     }

196     /**
197      * Checks if the disposal of the tiles on the board allows the player to end
198      * his turn, regarding the rules. The method needs to know that the game is
199      * in the first turn, or not, because that is a special case.
200      * 
201      * @param firstTurn
202      *            <code>true</code> if the game is in the first turn
203      * @return <code>true</code> if the state of the board matches the above
204      *         condition, <code>false</code> otherwise
205      */
206     public boolean isLegal(boolean firstTurn) {
207         // TODO debug
208         // for (int i = 0; i < BOARD_SIZE; i++) {
209         // for (int j = 0; j < BOARD_SIZE; j++) {
210         // System.out.format("%10s", board[i][j].getStatus());
211         // }
212         // System.out.println();
213         // }
214         // System.out.println();        
215 
216         int row = -1, col = -1;
217         boolean found = false;
218         for (int i = 0; i < BOARD_SIZE; i++) {
219             for (int j = 0; j < BOARD_SIZE; j++) {
220                 if (board[i][j].getStatus() == Field.Status.MOVABLE) {
221                     row = i;
222                     col = j;
223                     found = true;
224                     break;
225                 }
226             }
227             if (found)
228                 break;
229         }
230 
231         if (found) {
232             boolean connected = false;
233 
234             int tilesInRow = 0;
235             for (int j = col; j < BOARD_SIZE; j++) {
236                 if (board[row][j].getStatus() == Field.Status.MOVABLE) {
237                     tilesInRow++;
238                 }
239             }
240 
241             int tilesInCol = 0;
242             for (int i = row; i < BOARD_SIZE; i++) {
243 
244                 if (board[i][col].getStatus() == Field.Status.MOVABLE) {
245                     tilesInCol++;
246                 }
247             }
248             // TODO debug
249             // System.out.println("Row : " + row);
250             // System.out.println("Col: " + col);
251             //
252             // System.out.println("Tiles in row : " + tilesInRow);
253             // System.out.println("Tiles in col : " + tilesInCol);
254 
255             if (tilesInRow > 1) {
256                 if (tilesInRow < movableTiles) {
257                     return false;
258                 } else {
259                     int connectedTiles = 0;
260                     for (int j = col; j < BOARD_SIZE; j++) {
261                         if (board[row][j].getStatus() == Field.Status.EMPTY) {
262                             break;
263                         } else if (board[row][j].getStatus() == Field.Status.MOVABLE) {
264                             connectedTiles++;
265                         }
266                         if (!connected && !firstTurn) {
267                             if ((row > 0 && board[row - 1][j].getStatus() == Field.Status.FIXED)
268                                     || (row < BOARD_SIZE - 1 && board[row + 1][j]
269                                             .getStatus() == Field.Status.FIXED)
270                                     || (j > 0 && board[row][j - 1].getStatus() == Field.Status.FIXED)
271                                     || (j < BOARD_SIZE - 1 && board[row][j + 1]
272                                             .getStatus() == Field.Status.FIXED)) {
273                                 connected = true;
274                             }
275                         }
276                     }
277                     if (connectedTiles < movableTiles) {
278                         return false;
279                     }
280                 }
281 
282             } else if (tilesInCol > 1) {
283                 if (tilesInCol < movableTiles) {
284                     return false;
285                 } else {
286                     int connectedTiles = 0;
287                     for (int i = row; i < BOARD_SIZE; i++) {
288                         if (board[i][col].getStatus() == Field.Status.EMPTY) {
289                             break;
290                         } else if (board[i][col].getStatus() == Field.Status.MOVABLE) {
291                             connectedTiles++;
292                         }
293                         if (!connected && !firstTurn) {
294                             if ((col > 0 && board[i][col - 1].getStatus() == Field.Status.FIXED)
295                                     || (col < BOARD_SIZE - 1 && board[i][col + 1]
296                                             .getStatus() == Field.Status.FIXED)
297                                     || (i > 0 && board[i - 1][col].getStatus() == Field.Status.FIXED)
298                                     || (i < BOARD_SIZE - 1 && board[i + 1][col]
299                                             .getStatus() == Field.Status.FIXED)) {
300                                 connected = true;
301                             }
302                         }
303                     }
304                     if (connectedTiles < movableTiles) {
305                         return false;
306                     }
307                 }
308 
309             } else if (movableTiles > 1) {
310                 return false;
311 
312             } else {
313                 if (!connected && !firstTurn) {
314                     if ((col > 0 && board[row][col - 1].getStatus() == Field.Status.FIXED)
315                             || (col < BOARD_SIZE - 1 && board[row][col + 1]
316                                     .getStatus() == Field.Status.FIXED)
317                             || (row > 0 && board[row - 1][col].getStatus() == Field.Status.FIXED)
318                             || (row < BOARD_SIZE - 1 && board[row + 1][col]
319                                     .getStatus() == Field.Status.FIXED)) {
320                         connected = true;
321                     }
322                 }
323             }
324             if (!connected && !firstTurn) {
325                 return false;
326             }
327             if (firstTurn && board[7][7].getStatus() != Field.Status.MOVABLE) {
328                 return false;
329             }
330         } else {
331             return false;
332         }
333 
334         return true;
335     }

337     /**
338      * Reads a word horizontally, starting from the given position. It is called
339      * by the public method <code>getPlayedWords</code>.
340      * 
341      * @param row
342      *            the row index of the starting position
343      * @param col
344      *            the column index of the starting position
345      * @return the word read, or <code>null</code> if there is no tile on the
346      *         starting position
347      */
348     private String readRowWord(int row, int col) {
349         int startCol = 0;
350         StringBuilder word = new StringBuilder();
351 
352         if (board[row][col].getStatus() == Field.Status.EMPTY) {
353             return null;
354         }
355 
356         for (int j = col; j >= 0; j--) {
357             if (board[row][j].getStatus() == Field.Status.EMPTY) {
358                 startCol = j + 1;
359                 break;
360             }
361         }
362 
363         for (int j = startCol; j < BOARD_SIZE; j++) {
364             if (board[row][j].getStatus() == Field.Status.EMPTY) {
365                 break;
366             }
367             if (board[row][j].getTileCode() != Letters.JOKER_CODE) {
368                 word.append(Letters.getL(board[row][j].getTileCode()));
369             } else {
370                 word.append(Letters.getL(board[row][j].getJokerTileCode()));
371             }            
372         }
373 
374         return word.toString();
375     }

417     /**
418      * Returns a list of the words played in the current turn. The method
419      * assumes that the state of the board allows the player to end his turn
420      * regarding the rules, i.e. the <code>isLegal</code> method returns
421      * <code>true</code>, otherwise the return value will be undefined.
422      * 
423      * @return a list of strings containing the words played in the current
424      *         turn, or <code>null</code> if there were no words played in the
425      *         current turn
426      */
427     public List<String> getPlayedWords() {
428 
429         int col = -1, row = -1;
430         List<String> playedWords = new ArrayList<String>();
431 
432         boolean found = false;
433         for (int i = 0; i < BOARD_SIZE; i++) {
434             for (int j = 0; j < BOARD_SIZE; j++) {
435                 if (board[i][j].getStatus() == Field.Status.MOVABLE) {
436                     row = i;
437                     col = j;
438                     found = true;
439                     break;
440                 }
441             }
442             if (found) {
443                 break;
444             }
445         }
446 
447         if (!found) {
448             return null;
449         }
450 
451         boolean tilesInRow = false;
452         for (int j = col + 1; j < BOARD_SIZE; j++) {
453             if (board[row][j].getStatus() == Field.Status.MOVABLE) {
454                 tilesInRow = true;
455                 break;
456             }
457         }
458         boolean tilesInCol = false;
459         for (int i = row + 1; i < BOARD_SIZE; i++) {
460             if (board[i][col].getStatus() == Field.Status.MOVABLE) {
461                 tilesInCol = true;
462                 break;
463             }
464         }
465 
466         if (tilesInRow) {
467             playedWords.add(readRowWord(row, col));
468             for (int j = col; j < BOARD_SIZE; j++) {
469                 if (board[row][j].getStatus() == Field.Status.EMPTY) {
470                     break;
471                 } else if (board[row][j].getStatus() == Field.Status.MOVABLE) {
472                     String w = readColWord(row, j);
473                     if (w.length() > 1
474                             && !(w.equals("CS") || w.equals("GY")
475                                     || w.equals("LY") || w.equals("NY")
476                                     || w.equals("SZ") || w.equals("TY") || w
477                                         .equals("ZS"))) {
478                         playedWords.add(w);
479                     }
480                 }
481             }
482         } else if (tilesInCol) {
483             playedWords.add(readColWord(row, col));
484             for (int i = row; i < BOARD_SIZE; i++) {
485                 if (board[i][col].getStatus() == Field.Status.EMPTY) {
486                     break;
487                 } else if (board[i][col].getStatus() == Field.Status.MOVABLE) {
488                     String w = readRowWord(i, col);
489                     if (w.length() > 1
490                             && !(w.equals("CS") || w.equals("GY")
491                                     || w.equals("LY") || w.equals("NY")
492                                     || w.equals("SZ") || w.equals("TY") || w
493                                         .equals("ZS"))) {
494                         playedWords.add(w);
495                     }
496                 }
497             }
498         } else {
499             String rowWord = readRowWord(row, col);
500             if (rowWord.length() > 1
501                     && !(rowWord.equals("CS") || rowWord.equals("GY")
502                             || rowWord.equals("LY") || rowWord.equals("NY")
503                             || rowWord.equals("SZ") || rowWord.equals("TY") || rowWord
504                                 .equals("ZS"))) {
505                 playedWords.add(rowWord);
506             }
507             String colWord = readColWord(row, col);
508             if (colWord.length() > 1
509                     && !(colWord.equals("CS") || colWord.equals("GY")
510                             || colWord.equals("LY") || colWord.equals("NY")
511                             || colWord.equals("SZ") || colWord.equals("TY") || colWord
512                                 .equals("ZS"))) {
513                 playedWords.add(colWord);
514             }
515         }
516 
517         return playedWords;
518     }

520     /**
521      * Calculates the value in points of a word placed horizontally and
522      * including the given position. This method is used by the public
523      * <code>calcPoints()</code> method.
524      * 
525      * @param row
526      *            the row index of the starting position of the word
527      * @param col
528      *            the column index of the starting position of the word
529      * @return the value of the word in points
530      */
531     private int calcRowWordPts(int row, int col) {
532         int startCol = 0;
533         int pts = 0;
534         int factor = 1;
535 
536         if (board[row][col].getStatus() == Field.Status.EMPTY) {
537             return 0;
538         }
539 
540         // Find factors.
541         for (int j = col; j < BOARD_SIZE; j++) {
542             if (board[row][j].getStatus() == Field.Status.EMPTY) {
543                 break;
544             } else if (board[row][j].getStatus() == Field.Status.MOVABLE) {
545                 if (board[row][j].getMultiplier() == Field.Multiplier.DWORD) {
546                     factor *= 2;
547                 } else if (board[row][j].getMultiplier() == Field.Multiplier.TWORD) {
548                     factor *= 3;
549                 }
550             }
551         }
552 
553         // Find the first tile of the word.
554         for (int j = col; j >= 0; j--) {
555             if (board[row][j].getStatus() == Field.Status.EMPTY) {
556                 startCol = j + 1;
557                 break;
558             }
559         }
560 
561         // Calculate points.
562         for (int j = startCol; j < BOARD_SIZE; j++) {
563             if (board[row][j].getStatus() == Field.Status.EMPTY) {
564                 break;
565             } else if (board[row][j].getStatus() == Field.Status.MOVABLE) {
566                 switch (board[row][j].getMultiplier()) {
567                 case DLETTER:
568                     pts += 2 * factor
569                             * Letters.getValue(board[row][j].getTileCode());
570                     break;
571                 case TLETTER:
572                     pts += 3 * factor
573                             * Letters.getValue(board[row][j].getTileCode());
574                     break;
575                 default:
576                     pts += factor
577                             * Letters.getValue(board[row][j].getTileCode());
578                     break;
579                 }
580             } else {
581                 pts += factor * Letters.getValue(board[row][j].getTileCode());
582             }
583         }
584 
585         return pts;
586     }

623             if (board[i][col].getStatus() == Field.Status.EMPTY) {
624                 startRow = i + 1;
625                 break;
626             }
627         }
628 
629         // Calculate points.
630         for (int i = startRow; i < BOARD_SIZE; i++) {
631             if (board[i][col].getStatus() == Field.Status.EMPTY) {
632                 break;
633             } else if (board[i][col].getStatus() == Field.Status.MOVABLE) {
634                 switch (board[i][col].getMultiplier()) {
635                 case DLETTER:
636                     pts += 2 * factor
637                             * Letters.getValue(board[i][col].getTileCode());
638                     break;
639                 case TLETTER:
640                     pts += 3 * factor
641                             * Letters.getValue(board[i][col].getTileCode());
642                     break;
643                 default:
644                     pts += factor
645                             * Letters.getValue(board[i][col].getTileCode());
646                     break;
647                 }
648             } else {
649                 pts += factor * Letters.getValue(board[i][col].getTileCode());
650             }
651         }
652 
653         return pts;
654     }

656     /**
657      * Returns the number of points the player will get in the current turn,
658      * based on the current state of the board. The method assumes that the
659      * state of the board allows the player to end his turn regarding the rules,
660      * i.e. the <code>isLegal()</code> method returns <code>true</code>,
661      * otherwise the return value will be undefined.
662      * 
663      * @return the total number of points gained in the current turn, or
664      *         undefined value if the <code>isLegal()</code> method returns
665      *         <code>false</code>
666      */
667     public int calcPoints() {
668         int pts = 0;
669         int col = -1, row = -1; // if these remain -1, something's wrong with
670         // the isLegal method
671 
672         boolean found = false;
673         for (int i = 0; i < BOARD_SIZE; i++) {
674             for (int j = 0; j < BOARD_SIZE; j++) {
675                 if (board[i][j].getStatus() == Field.Status.MOVABLE) {
676                     row = i;
677                     col = j;
678                     found = true;
679                     break;
680                 }
681             }
682             if (found) {
683                 break;
684             }
685         }
686 
687         if (!found) {
688             return 0;
689         }
690 
691         boolean tilesInRow = false;
692         for (int j = col + 1; j < BOARD_SIZE; j++) {
693             if (board[row][j].getStatus() == Field.Status.MOVABLE) {
694                 tilesInRow = true;
695                 break;
696             }
697         }
698         boolean tilesInCol = false;
699         for (int i = row + 1; i < BOARD_SIZE; i++) {
700             if (board[i][col].getStatus() == Field.Status.MOVABLE) {
701                 tilesInCol = true;
702                 break;
703             }
704         }
705 
706         if (tilesInRow) {
707             pts += calcRowWordPts(row, col);
708             for (int j = col; j < BOARD_SIZE; j++) {
709                 if (board[row][j].getStatus() == Field.Status.EMPTY) {
710                     break;
711                 } else if (board[row][j].getStatus() == Field.Status.MOVABLE) {
712                     if ((row != 0 && board[row - 1][j].getStatus() != Field.Status.EMPTY)
713                             || (row < BOARD_SIZE - 1 && board[row + 1][j]
714                                     .getStatus() != Field.Status.EMPTY)) {
715                         pts += calcColWordPts(row, j);
716                     }
717                 }
718             }
719         } else if (tilesInCol) {
720             pts += calcColWordPts(row, col);
721             for (int i = row; i < BOARD_SIZE; i++) {
722                 if (board[i][col].getStatus() == Field.Status.EMPTY) {
723                     break;
724                 } else if (board[i][col].getStatus() == Field.Status.MOVABLE) {
725                     if ((col != 0 && board[i][col - 1].getStatus() != Field.Status.EMPTY)
726                             || (col < BOARD_SIZE - 1 && board[i][col + 1]
727                                     .getStatus() != Field.Status.EMPTY)) {
728                         pts += calcRowWordPts(i, col);
729                     }
730                 }
731             }
732         } else {
733             if (board[row][col].getStatus() == Field.Status.MOVABLE) {
734                 if ((row != 0 && board[row - 1][col].getStatus() != Field.Status.EMPTY)
735                         || (row < BOARD_SIZE - 1 && board[row + 1][col]
736                                 .getStatus() != Field.Status.EMPTY)) {
737                     pts += calcColWordPts(row, col);
738                 }
739                 if ((col != 0 && board[row][col - 1].getStatus() != Field.Status.EMPTY)
740                         || (col < BOARD_SIZE - 1 && board[row][col + 1]
741                                 .getStatus() != Field.Status.EMPTY)) {
742                     pts += calcRowWordPts(row, col);
743                 }
744             }
745         }
746 
747         return pts;
748     }

777     /**
778      * Checks if the tiles placed on the board in the current turn form
779      * grammatically correct words. If there are no such tiles, the method will
780      * return <code>false</code>. The method receives a {@link Dictionary}
781      * object with the words considered correct.
782      * 
783      * @param dict
784      *            the <code>Dictionary</code> object
785      * @return <code>true</code> if all the words placed on the board in the
786      *         current turn are correct, <code>false</code> if there are no
787      *         tiles placed on the board in the current turn, or there is at
788      *         least one grammatically incorrect word.
789      */
790     public boolean wordsAreCorrect(Dictionary dict) {
791         // System.out.println("Movable tiles: " + movableTiles);
792         List<String> playedWords = getPlayedWords();
793         if (playedWords == null) {
794             return false;
795         }
796         boolean areCorrect = true;
797 
798         for (String word : playedWords) {
799             if (!dict.isCorrect(word)) {
800                 areCorrect = false;
801                 break;
802             }
803         }
804 
805         return areCorrect;
806     }
